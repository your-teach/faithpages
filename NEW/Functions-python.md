# Функции Python

## Функции
```python
# Объявление функции
def func():
    print('Помахать рукой')
    print('Подойти к человеку')
    print('Пожать руку')
    print('Приветствие:')
    print('Здравствуйте, человек!')

func() # Запуск функции
func()
print(func) # объект функции, не является запуском

# Функции с аргументами
def func(name):
    print('Помахать рукой')
    print('Подойти к ' + name)
    print('Пожать руку')
    print('Приветствие:')
    print('Здравствуйте, ' + name)

func(name='Федор')
func(name='Александр')
```
