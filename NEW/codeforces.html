Игра с кубиком#

https://codeforces.com/problemset/problem/378/A

nums = input()
nums = nums.split()
a, b = int(nums[0]), int(nums[1])
f = 0
t = 0
s = 0
for i in range(1, 7):
    if abs(a - i) < abs(b - i):
        f += 1
    elif abs(a - i) > abs(b - i):
        t += 1
    else:
        s += 1
print(f,s,t)
